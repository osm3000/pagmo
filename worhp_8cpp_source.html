73  // We set some of the parameters exposed in the constructor
74  set_param("TolFeas", feas);
75  set_param("TolOpti", opt);
76  set_param("MaxIter", iter);

86 void worhp::evolve(pagmo::population& pop) const {
87  if (pop.size() == 0) {
88  return;
89  }
90 
91  const auto& prob = pop.problem();
92 
93  if (prob.get_i_dimension() != 0) {
94  pagmo_throw(value_error,
95  "The problem has an integer part and WORHP is not suitable to solve it.");
96  }
97 
98  if (prob.get_f_dimension() != 1) {
99  pagmo_throw(value_error,
100  "The problem is not single objective and WORHP is not suitable to solve it");
101  }
102 
103  // We check the screen output again as the user may have changed it
104  if (m_screen_output) {
105  SetWorhpPrint(default_output);
106  } else {
107  SetWorhpPrint(no_screen_output);
108  }
109 
110  OptVar opt;
111  Control control;
112  Params params;
113  Workspace workspace;
114 
115  opt.initialised = false;
116  control.initialised = false;
117  params.initialised = false;
118  workspace.initialised = false;
119 
120 
121  opt.n = prob.get_dimension(); // number of variables
122  opt.m = prob.get_c_dimension(); // number of constraints
123  auto n_eq = prob.get_c_dimension() - prob.get_ic_dimension(); // number of equality constraints
124 
125  // specify nonzeros of derivative matrixes
126  workspace.DF.nnz = opt.n; // dense
127  workspace.DG.nnz = opt.n * opt.m; // dense
128  workspace.HM.nnz = opt.n;
129 
130 
131  WorhpInit(&opt, &workspace, &params, &control);
132  assert(control.status == FirstCall);
133  params = m_params;
134 
135  // Specify a derivative free case
136  params.UserDF = false;
137  params.UserDG = false;
138  params.UserHM = false;
139  params.initialised = true;
140 
141  // Initialization of variables
142  const auto best_idx = pop.get_best_idx();
143  pagmo::decision_vector x = pop.get_individual(best_idx).cur_x;
144 
145  for (int i = 0; i < opt.n; ++i) {
146  opt.X[i] = x[i];
147  opt.Lambda[i] = 0;
148  opt.XL[i] = prob.get_lb()[i];
149  opt.XU[i] = prob.get_ub()[i];
150  }
151 
152  // Equality constraints
153  for (auto i = 0u; i < n_eq; ++i) {
154  opt.Mu[i] = 0;
155  opt.GL[i] = 0;
156  opt.GU[i] = 0;
157  }
158 
159  // Inequality constraints
160  for (auto i = n_eq; i < unsigned(opt.m); ++i) {
161  opt.Mu[i] = 0;
162  opt.GL[i] = -params.Infty;
163  opt.GU[i] = 0;
164  }
165 
166  // Define HM as a diagonal LT-CS-matrix, but only if needed by WORHP
167  // Not sure if this is needed at all
168  if (workspace.HM.NeedStructure) {
169  for(int i = 0; i < workspace.HM.nnz; ++i)
170  {
171  workspace.HM.row[i] = i + 1;
172  workspace.HM.col[i] = i + 1;
173  }
174  }
175 
176  while (control.status < TerminateSuccess && control.status > TerminateError) {
177  if (GetUserAction(&control, callWorhp)) {
178  Worhp(&opt, &workspace, &params, &control);
179  }
180 
181  if (GetUserAction(&control, iterOutput))
182  {
183  IterationOutput(&opt, &workspace, &params, &control);
184  DoneUserAction(&control, iterOutput);
185  }
186 
187 
188  if (GetUserAction(&control, evalF)) {
189  for (int i = 0; i < opt.n; ++i) {
190  x[i] = opt.X[i];
191  }
192  auto f = prob.objfun(x);
193  opt.F = workspace.ScaleObj * f[0];
194  DoneUserAction(&control, evalF);
195  }
196 
197  if (GetUserAction(&control, evalG)) {
198  for (int i = 0; i < opt.n; ++i) {
199  x[i] = opt.X[i];
200  }
201  auto g = prob.compute_constraints(x);
202  for (int i = 0; i < opt.m; ++i) {
203  opt.G[i] = g[i];
204  }
205  DoneUserAction(&control, evalG);
206  }
207 
208  if (GetUserAction(&control, fidif)) {
209  WorhpFidif(&opt, &workspace, &params, &control);
210  }
211  }
212 
213  for (int i = 0; i < opt.n; ++i) {
214  x[i] = opt.X[i];
215  }
216  pop.set_x(best_idx, x);
217 
218  StatusMsg(&opt, &workspace, &params, &control);
219  WorhpFree(&opt, &workspace, &params, &control);
220 }
